122 void DragBoxMarker::UpdateArrow(const std::string& axis,
123  const std::string& polarity,
124  const geometry_msgs::Point& point) {
125  if (polarity == "pos") {
126  if (axis == "x") {
127  max_x_ = point.x - 0.05;
128  min_x_ = std::min(min_x_, max_x_ - 0.01);
129  } else if (axis == "y") {
130  max_y_ = point.y - 0.05;
131  min_y_ = std::min(min_y_, max_y_ - 0.01);
132  } else if (axis == "z") {
133  max_z_ = point.z - 0.05;
134  min_z_ = std::min(min_z_, max_z_ - 0.01);
135  }
136  } else if (polarity == "neg") {
137  if (axis == "x") {
138  min_x_ = point.x + 0.05;
139  max_x_ = std::max(max_x_, min_x_ + 0.01);
140  } else if (axis == "y") {
141  min_y_ = point.y + 0.05;
142  max_y_ = std::max(max_y_, min_y_ + 0.01);
143  } else if (axis == "z") {
144  min_z_ = point.z + 0.05;
145  max_z_ = std::max(max_z_, min_z_ + 0.01);
146  }
147  }
148  Update();
149 }

151 void DragBoxMarker::Feedback(
152  const InteractiveMarkerFeedback::ConstPtr& feedback) {
153  if (feedback->event_type != InteractiveMarkerFeedback::POSE_UPDATE) {
154  return;
155  }
156 
157  transform_graph::Position local_position;
158  tf_graph_.DescribePosition(feedback->pose.position,
159  transform_graph::Source(feedback->header.frame_id),
160  transform_graph::Target("box"), &local_position);
161  geometry_msgs::Point local_point;
162  local_point.x = local_position.vector().x();
163  local_point.y = local_position.vector().y();
164  local_point.z = local_position.vector().z();
165  if (feedback->marker_name == name_ + "-pos_x") {
166  UpdateArrow("x", "pos", local_point);
167  } else if (feedback->marker_name == name_ + "-pos_y") {
168  UpdateArrow("y", "pos", local_point);
169  } else if (feedback->marker_name == name_ + "-pos_z") {
170  UpdateArrow("z", "pos", local_point);
171  } else if (feedback->marker_name == name_ + "-neg_x") {
172  UpdateArrow("x", "neg", local_point);
173  } else if (feedback->marker_name == name_ + "-neg_y") {
174  UpdateArrow("y", "neg", local_point);
175  } else if (feedback->marker_name == name_ + "-neg_z") {
176  UpdateArrow("z", "neg", local_point);
177  }
178 }

180 void DragBoxMarker::MakeBox(InteractiveMarker* box) {
181  double x = (min_x_ + max_x_) / 2;
182  double y = (min_y_ + max_y_) / 2;
183  double z = (min_z_ + max_z_) / 2;
184  double dim_x = max_x_ - min_x_;
185  double dim_y = max_y_ - min_y_;
186  double dim_z = max_z_ - min_z_;
187  Marker marker;
188  marker.type = Marker::CUBE;
189  marker.pose.position.x = x;
190  marker.pose.position.y = y;
191  marker.pose.position.z = z;
192  marker.scale.x = dim_x;
193  marker.scale.y = dim_y;
194  marker.scale.z = dim_z;
195  marker.color.b = 1;
196  marker.color.a = 0.3;
197 
198  InteractiveMarkerControl control;
199  control.always_visible = true;
200  control.interaction_mode = InteractiveMarkerControl::NONE;
201  control.markers.push_back(marker);
202 
203  box->name = name_ + "-box";
204  box->header = pose_stamped_.header;
205  box->pose = pose_stamped_.pose;
206  box->scale = 1;
207  box->controls.clear();
208  box->controls.push_back(control);
209 }

211 void DragBoxMarker::MakeArrow(const std::string& axis,
212  const std::string& polarity,
213  visualization_msgs::InteractiveMarker* arrow) {
214  Marker marker;
215  marker.type = Marker::ARROW;
216  marker.scale.x = 0.05;
217  marker.scale.y = 0.025;
218  marker.scale.z = 0.025;
219  if (axis == "x") {
220  if (polarity == "pos") {
221  marker.color.r = 1;
222  } else {
223  marker.color.r = 0.5;
224  }
225  } else if (axis == "y") {
226  if (polarity == "pos") {
227  marker.color.g = 1;
228  } else {
229  marker.color.g = 0.5;
230  }
231  } else if (axis == "z") {
232  if (polarity == "pos") {
233  marker.color.b = 1;
234  } else {
235  marker.color.b = 0.5;
236  }
237  }
238  marker.color.a = 0.5;
239 
240  tf::Quaternion arrow_q;
241  geometry_msgs::Pose arrow_pose;
242  arrow_pose.position.x = (max_x_ + min_x_) / 2;
243  arrow_pose.position.y = (max_y_ + min_y_) / 2;
244  arrow_pose.position.z = (max_z_ + min_z_) / 2;
245  if (polarity == "pos") {
246  if (axis == "x") {
247  arrow_q = tf::Quaternion(0, 0, 0, 1);
248  arrow_pose.position.x = max_x_ + 0.05;
249  } else if (axis == "y") {
250  arrow_q = tf::Quaternion(0, 0, 1, 1);
251  arrow_pose.position.y = max_y_ + 0.05;
252  } else if (axis == "z") {
253  arrow_q = tf::Quaternion(0, -1, 0, 1);
254  arrow_pose.position.z = max_z_ + 0.05;
255  }
256  } else if (polarity == "neg") {
257  if (axis == "x") {
258  arrow_q = tf::Quaternion(0, 0, 1, 0);
259  arrow_pose.position.x = min_x_ - 0.05;
260  } else if (axis == "y") {
261  arrow_q = tf::Quaternion(0, 0, -1, 1);
262  arrow_pose.position.y = min_y_ - 0.05;
263  } else if (axis == "z") {
264  arrow_q = tf::Quaternion(0, 1, 0, 1);
265  arrow_pose.position.z = min_z_ - 0.05;
266  }
267  }
268  arrow_q.normalize();
269  tf::quaternionTFToMsg(arrow_q, marker.pose.orientation);
270 
271  InteractiveMarkerControl control;
272  control.orientation.w = 1;
273 
274  double sign = 1;
275  if (polarity == "neg") {
276  sign = -1;
277  }
278  if (axis == "x") {
279  if (sign == -1) {
280  control.orientation.w = 0;
281  control.orientation.z = 1;
282  }
283  } else if (axis == "y") {
284  control.orientation.z = sign;
285  } else if (axis == "z") {
286  control.orientation.y = -sign;
287  }
288  control.interaction_mode = InteractiveMarkerControl::MOVE_AXIS;
289  control.always_visible = true;
290  control.markers.push_back(marker);
291 
292  transform_graph::Transform global_arrow_pose_tf;
293  tf_graph_.DescribePose(arrow_pose, transform_graph::Source("box"),
294  transform_graph::Target(pose_stamped_.header.frame_id),
295  &global_arrow_pose_tf);
296 
297  arrow->name = name_ + "-" + polarity + "_" + axis;
298  arrow->header.frame_id = pose_stamped_.header.frame_id;
299  global_arrow_pose_tf.ToPose(&arrow->pose);
300  arrow->controls.push_back(control);
301  arrow->scale = 0.05;
302 }
