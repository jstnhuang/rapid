175 void GroupingPoseEstimator::FindCorrespondences(
176  pcl::CorrespondencesPtr model_scene_corrs) {
177  pcl::KdTreeFLANN<SHOT352> match_search;
178  match_search.setInputCloud(object_descriptors_);
179  // For each scene keypoint descriptor, find nearest neighbor into the model
180  // keypoints descriptor cloud and add it to the correspondences vector.
181  int nan_count = 0;
182  int no_neighbors = 0;
183  int under_threshold = 0;
184  visualization_msgs::MarkerArray markers;
185  for (size_t i = 0; i < scene_descriptors_->size(); ++i) {
186  std::vector<int> neigh_indices(1);
187  std::vector<float> neigh_sqr_dists(1);
188  if (!pcl_isfinite(scene_descriptors_->at(i).descriptor[0])) {
189  ++nan_count;
190  continue;
191  }
192  int found_neighs = match_search.nearestKSearch(
193  scene_descriptors_->at(i), 1, neigh_indices, neigh_sqr_dists);
194  if (found_neighs < 1) {
195  ++no_neighbors;
196  continue;
197  }
198  // add match only if the squared descriptor distance is less than 0.25
199  // (SHOT descriptor distances are between 0 and 1 by design)
200  if (neigh_sqr_dists[0] < corr_match_threshold_) {
201  pcl::Correspondence corr(neigh_indices[0], static_cast<int>(i),
202  neigh_sqr_dists[0]);
203  model_scene_corrs->push_back(corr);
204  } else {
205  ++under_threshold;
206  }
207  }
208  // Visualize correspondences
209  for (size_t i = 0; i < model_scene_corrs->size(); ++i) {
210  int index_query = model_scene_corrs->at(i).index_query;
211  int index_match = model_scene_corrs->at(i).index_match;
212 
213  const PointC& model_point = object_keypoints_->at(index_query);
214  const PointC& scene_point = scene_keypoints_->at(index_match);
215  visualization_msgs::Marker marker;
216  marker.header.frame_id = "base_footprint";
217  marker.ns = "corrs";
218  marker.id = i;
219  marker.type = visualization_msgs::Marker::ARROW;
220  marker.scale.x = 0.0025;
221  marker.scale.y = 0.005;
222  marker.color.r = 1;
223  marker.color.a = 0.5;
224  marker.points.resize(2);
225  marker.points[0].x = model_point.x;
226  marker.points[0].y = model_point.y;
227  marker.points[0].z = model_point.z;
228  marker.points[1].x = scene_point.x;
229  marker.points[1].y = scene_point.y;
230  marker.points[1].z = scene_point.z;
231  markers.markers.push_back(marker);
232  }
233 
234  correspondence_pub_.publish(markers);
235 
236  ROS_INFO("Found %ld correspondences", model_scene_corrs->size());
237  ROS_INFO(
238  "Out of %ld keypoints, %d were NaNs, %d had no neighbors, and %d were "
239  "over the threshold %f",
240  scene_descriptors_->size(), nan_count, no_neighbors, under_threshold,
241  corr_match_threshold_);
242 }
