46 bool ParseScene(const PointCloud<PointXYZRGB>::ConstPtr& cloud,
47  const ParseParams& params, Scene* scene) {
48  // This function just crops the scene to a box given by the params.
49  scene->set_cloud(cloud);
50  Eigen::Vector4f min;
51  min << params.scene.min_x, params.scene.min_y, params.scene.min_z, 1;
52  Eigen::Vector4f max;
53  max << params.scene.max_x, params.scene.max_y, params.scene.max_z, 1;
54  pcl::CropBox<PointXYZRGB> cb;
55  cb.setInputCloud(cloud);
56  cb.setMin(min);
57  cb.setMax(max);
58 
59  vector<int> box_indices;
60  cb.filter(box_indices);
61 
62  PointIndices::Ptr pcl_indices(new PointIndices);
63  pcl_indices->indices = box_indices;
64  HSurface primary_surface;
65  bool success = ParseHSurface(cloud, pcl_indices, params, &primary_surface);
66  if (!success) {
67  return false;
68  }
69  scene->set_primary_surface(primary_surface);
70  geometry_msgs::PoseStamped ps;
71  ps.header.frame_id = cloud->header.frame_id;
72  ps.pose.position.x = (params.scene.max_x + params.scene.min_x) / 2;
73  ps.pose.position.y = (params.scene.max_y + params.scene.min_y) / 2;
74  ps.pose.position.z = (params.scene.max_z + params.scene.min_z) / 2;
75  ps.pose.orientation.w = 1;
76  scene->set_pose(ps);
77  geometry_msgs::Vector3 scale;
78  scale.x = params.scene.max_x - params.scene.min_x;
79  scale.y = params.scene.max_y - params.scene.min_y;
80  scale.z = params.scene.max_z - params.scene.min_z;
81  scene->set_scale(scale);
82  return true;
83 }

85 bool ParseHSurface(const pcl::PointCloud<pcl::PointXYZRGB>::ConstPtr& cloud,
86  const PointIndices::ConstPtr& indices,
87  const ParseParams& params, HSurface* surface) {
88  // Find the dominant plane, which should be mostly horizontal. This becomes
89  // the HSurface.
90  PointIndices::Ptr plane_inliers(new PointIndices);
91  bool found =
92  FindHorizontalPlane(cloud, indices, params.hsurface.distance_threshold,
93  params.hsurface.eps_angle, plane_inliers);
94  if (!found) {
95  return false;
96  }
97 
98  geometry_msgs::PoseStamped ps;
99  geometry_msgs::Vector3 scale;
100  GetBoundingBox(cloud, plane_inliers, &ps, &scale);
101 
102  surface->SetCloud(cloud, plane_inliers);
103  surface->set_name("primary_surface");
104  surface->set_pose(ps);
105  surface->set_scale(scale);
106 
107  // Find points above the HSurface
108  PointIndices::Ptr object_indices(new PointIndices);
109  tf::Vector3 surface_translation(ps.pose.position.x, ps.pose.position.y,
110  ps.pose.position.z);
111  tf::Quaternion surface_orientation;
112  tf::quaternionMsgToTF(ps.pose.orientation, surface_orientation);
113  tf::Transform base_to_surface(surface_orientation, surface_translation);
114  for (size_t i = 0; i < indices->indices.size(); ++i) {
115  int index = indices->indices[i];
116  const PointXYZRGB& point = cloud->points[index];
117  tf::Vector3 point_v(point.x, point.y, point.z);
118  // Project point into table space
119  tf::Vector3 table_p = base_to_surface.inverse() * point_v;
120  if (table_p.x() >= -scale.x / 2 && table_p.x() <= scale.x / 2 &&
121  table_p.y() >= -scale.y / 2 && table_p.y() <= scale.y / 2 &&
122  table_p.z() > scale.z / 2) {
123  object_indices->indices.push_back(index);
124  }
125  }
126 
127  // We must reify the object cloud now because the outlier filters search for
128  // neighbors in the entire input cloud, not just in the indices subset.
129  PointCloud<PointXYZRGB>::Ptr object_cloud =
130  IndicesToCloud(cloud, object_indices);
131 
132  pcl::RadiusOutlierRemoval<PointXYZRGB> radius;
133  radius.setInputCloud(object_cloud);
134  radius.setRadiusSearch(0.05);
135  radius.setMinNeighborsInRadius(10);
136  radius.filter(*object_cloud);
137 
138  vector<Object> objects;
139  PointIndices::Ptr all_indices(new PointIndices);
140  ParseObjects(object_cloud, all_indices, params, &objects);
141  for (size_t i = 0; i < objects.size(); ++i) {
142  surface->AddObject(objects[i]);
143  }
144  return true;
145 }

147 bool ParseObjects(const pcl::PointCloud<pcl::PointXYZRGB>::ConstPtr& cloud,
148  const pcl::PointIndices::ConstPtr& indices,
149  const ParseParams& params, std::vector<Object>* objects) {
150  pcl::search::KdTree<PointXYZRGB>::Ptr tree(
151  new pcl::search::KdTree<PointXYZRGB>);
152  tree->setInputCloud(cloud);
153 
154  vector<PointIndices> cluster_indices;
155  pcl::RegionGrowingRGB<PointXYZRGB> ec;
156  ec.setInputCloud(cloud);
157  if (indices->indices.size() != 0) {
158  ec.setIndices(indices);
159  }
160  ec.setSearchMethod(tree);
161  ec.setDistanceThreshold(params.objects.distance_threshold);
162  ec.setPointColorThreshold(params.objects.point_color_threshold);
163  ec.setRegionColorThreshold(params.objects.region_color_threshold);
164  ec.setMinClusterSize(params.objects.min_cluster_size);
165  ec.extract(cluster_indices);
166 
167  int obj_num = 0;
168  for (vector<PointIndices>::const_iterator it = cluster_indices.begin();
169  it != cluster_indices.end(); ++it) {
170  PointIndices::Ptr cluster_indices_p(new PointIndices);
171  *cluster_indices_p = *it;
172  geometry_msgs::PoseStamped ps;
173  ps.header.frame_id = cloud->header.frame_id;
174  geometry_msgs::Vector3 scale;
175  GetPlanarBoundingBox(cloud, cluster_indices_p, &ps.pose, &scale);
176  std::stringstream ss;
177  ss << "object_" << obj_num++;
178 
179  Object obj;
180  obj.SetCloud(cloud, cluster_indices_p);
181  obj.set_name(ss.str());
182  obj.set_pose(ps);
183  obj.set_scale(scale);
184  objects->push_back(obj);
185  }
186  return true;
187 }
