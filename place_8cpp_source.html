124 bool SampleRandomPlacement(double obj_height, const HSurface& table,

128  geometry_msgs::PoseStamped table_ps = table.pose();

130  double start_x = -table.scale().x / 2 + distance_from_edge;
131  double y_range = table.scale().y - 2 * distance_from_edge;

133  double z = table.scale().z / 2 + obj_height / 2;
134

136  tf::Vector3 table_origin(table_ps.pose.position.x, table_ps.pose.position.y,
137  table_ps.pose.position.z);
138  tf::Quaternion table_orientation(
139  table_ps.pose.orientation.x, table_ps.pose.orientation.y,
140  table_ps.pose.orientation.z, table_ps.pose.orientation.w);
141  table_frame.setOrigin(table_origin);
142  table_frame.setRotation(table_orientation);
143 
144  int max_tries = 0;
145  ros::param::param<int>("sample_placement_max_tries", max_tries, 100);
146  double obstacle_distance = 0;
147  ros::param::param<double>("obstacle_distance", obstacle_distance, 0.1);
148  double obs_dist_squared = obstacle_distance * obstacle_distance;
149  for (int i = 0; i < max_tries; ++i) {
150  // Sample in table space
151  tf::Vector3 pos;
152  pos.setX(start_x + static_cast<double>(std::rand()) / RAND_MAX * x_range);
153  pos.setY(start_y + static_cast<double>(std::rand()) / RAND_MAX * y_range);
154  pos.setZ(z);
155  tf::Vector3 transformed = table_frame * pos;
156  ROS_INFO("Try %d (base frame): x=%f, y=%f, z=%f", i, transformed.x(),
157  transformed.y(), transformed.z());
158 
159  // Check that the object doesn't intersect with all other objects.
160  // TODO(jstn): do a real check of object intersection. For now, just check
161  // that it's some distance away from other objects.
162  // TODO(jstn): this assumes objects are in the base frame.
163  bool intersect = false;
164  const vector<Object>& objects = table.objects();
165  for (size_t obj_i = 0; obj_i < objects.size(); ++obj_i) {

167  double xd = obj_pos.x - transformed.x();
168  double yd = obj_pos.y - transformed.y();
169  double squared_distance = xd * xd + yd * yd;
170  intersect = squared_distance < obs_dist_squared;
171  if (intersect) {

173  }
174  }
