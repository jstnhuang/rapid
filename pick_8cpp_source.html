126 bool Picker::Pick(const Object& obj, double max_effort) {
127  // Use agile_grasp to find some grasp poses.
128  Grasps grasps_response;
129  bool success = GenerateGrasps(obj, &grasps_response);
130  if (!success) {
131  ROS_ERROR("Grasp generation service failed.");
132  return false;
133  }
134  vector<Grasp>& grasps = grasps_response.grasps;
135  if (grasps.size() == 0) {
136  ROS_ERROR("Failed to find viable grasps.");
137  return false;
138  }
139 
140  std::string frame = grasps_response.header.frame_id;
141  if (frame == "") {
142  // This must always be the camera frame passed into agile_grasp.
143  frame = "/head_mount_kinect_rgb_optical_frame";
144  }
145 
146  // TODO(jstn): better grasp selection
147  std::random_shuffle(grasps.begin(), grasps.end());
148  int num_grasps = 0;
149  ros::param::param<int>("num_grasps", num_grasps, 10);
150 
151  bool grasped_object = false;
152  for (size_t grasp_i = 0;
153  grasp_i < std::min(static_cast<size_t>(num_grasps), grasps.size());
154  ++grasp_i) {
155  const Grasp& grasp = grasps[grasp_i];
156  VisualizeGrasp(grasp, frame);
157 
158  PoseStamped grasp_pose;
159  grasp_pose.header.frame_id = frame;
160  grasp_pose.pose.position.x = grasp.surface_center.x;
161  grasp_pose.pose.position.y = grasp.surface_center.y;
162  grasp_pose.pose.position.z = grasp.surface_center.z;
163  ComputeGraspOrientation(grasp.approach, grasp.axis,
164  &grasp_pose.pose.orientation);
165 
166  // Pre-grasp
167  ROS_INFO("Attempting pre-grasp");
168  success = MoveToPreGrasp(grasp_pose, grasp.approach);
169  if (!success) {
170  ROS_ERROR("Failed to move to pre-grasp position.");
171  continue;
172  }
173  gripper_->Open();
174 
175  // Grasp
176  ROS_INFO("Attempting grasp");
177  success = MoveToGrasp(grasp_pose, grasp.approach);
178  if (!success) {
179  ROS_ERROR("Failed to execute grasp.");
180  continue;
181  }
182  gripper_->Close();
183  gripper_->set_held_object(obj);
184  grasped_object = true;
185  break;
186  }
187 
188  if (!grasped_object) {
189  ROS_ERROR("Failed to grasp object.");
190  return false;
191  }
192 
193  ROS_INFO("Pick succeeded");
194  return true;
195 }

212  const Vector3& grasp_approach) {
213  PoseStamped pre_grasp = grasp_pose;
214  double pregrasp_distance = 0;

254 void ComputeGraspOrientation(const geometry_msgs::Vector3& approach,
255  const geometry_msgs::Vector3& axis,
256  geometry_msgs::Quaternion* orientation) {
257  // Not sure if grasp.axis and grasp.approach are always orthogonal, so
258  // compute y = x cross z, then z = x cross y.
259  Eigen::Vector3d e_approach;
260  e_approach << approach.x, approach.y, approach.z;
261  Eigen::Vector3d e_axis;
262  e_axis << axis.x, axis.y, axis.z;
263  Eigen::Vector3d grasp_y;
264  grasp_y = e_axis.cross(e_approach);
265  e_axis = e_approach.cross(grasp_y);
266 
267  Eigen::Matrix3d rotation;
268  rotation.col(0) = e_approach;
269  rotation.col(1) = grasp_y;
270  rotation.col(2) = e_axis;
271 
272  Eigen::Quaterniond eigen_q(rotation);
273  eigen_q.normalize();
274  geometry_msgs::Quaternion grasp_q;
275  orientation->w = eigen_q.w();
276  orientation->x = eigen_q.x();
277  orientation->y = eigen_q.y();
278  orientation->z = eigen_q.z();
279 }
