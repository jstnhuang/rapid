264 void PoseEstimator::RunIcpCandidateInThread(
265  PointCloudC::Ptr candidates, size_t candidate_index, int max_iterations,
266  boost::mutex& output_mutex, vector<PoseEstimationMatch>* aligned_objects) {
267  // Copy the object and its center.
268  PointCloudC::Ptr working_object(new PointCloudC);
269  PointCloudC::Ptr aligned_object(new PointCloudC);
270  pcl::IterativeClosestPoint<PointC, PointC> icp;
271  icp.setInputTarget(scene_);
272  icp.setMaximumIterations(max_iterations);
273 
274  // Compute object to base transform.
275  Eigen::Affine3f object_to_base;
276  object_to_base = Eigen::Translation3f(-object_center_);
277 
278  pcl::PointXYZ roi_translation;
279  utils::GeometryMsgToPcl(object_roi_.transform.translation, &roi_translation);
280  Eigen::Quaternionf roi_rotation;
281  utils::GeometryMsgToEigen(object_roi_.transform.rotation, &roi_rotation);
282 
283  const PointC& center = candidates->at(candidate_index);
284  rapid_msgs::Roi3D roi = object_roi_;
285 
286  Eigen::Affine3f base_to_candidate;
287  base_to_candidate = Eigen::Translation3f(center.x, center.y, center.z);
288 
289  // Transform the object to be centered on the center point
290  // Also assume that the object and scene are in the same frame
291 
292  vector<Eigen::Quaternionf> rotations;
293  GenerateRotations(&rotations);
294  for (size_t rot_i = 0; rot_i < rotations.size(); ++rot_i) {
295  Eigen::Affine3f candidate_rotation = Eigen::Affine3f::Identity();
296  candidate_rotation *= rotations[rot_i];
297 
298  Eigen::Affine3f candidate_transform =
299  base_to_candidate * candidate_rotation * object_to_base;
300  pcl::transformPointCloud(*object_, *working_object, candidate_transform);
301  // if (debug_) {
302  // viz::PublishCloud(alignment_pub_, *working_object);
303  // std::cout << "Showing candidate object, press enter to continue";
304  // string input;
305  // std::getline(std::cin, input);
306  //}
307 
308  // Run ICP
309  icp.setInputSource(working_object);
310  icp.align(*aligned_object);
311  viz::PublishCloud(alignment_pub_, *aligned_object);
312  Eigen::Matrix4f icp_transform = icp.getFinalTransformation();
313  Eigen::Affine3f icp_affine;
314  icp_affine = icp_transform;
315  Eigen::Affine3f final_affine = icp_affine * candidate_transform;
316 
317  // Apply transform to ROI position
318  pcl::PointXYZ roi_pt = pcl::transformPoint(roi_translation, final_affine);
319  utils::PclToGeometryMsg(roi_pt, &roi.transform.translation);
320 
321  // Apply transform to ROI rotation
322  Eigen::Affine3f final_rotation = final_affine * roi_rotation;
323  Eigen::Quaternionf q;
324  q = final_rotation.rotation();
325  utils::EigenToGeometryMsg(q, &roi.transform.rotation);
326 
327  // The transformation of the match, in the base frame.
328  double fitness = ComputeIcpFitness(scene_, aligned_object, roi);
329  geometry_msgs::Pose output_pose;
330  output_pose.position.x = roi.transform.translation.x;
331  output_pose.position.y = roi.transform.translation.y;
332  output_pose.position.z = roi.transform.translation.z;
333  output_pose.orientation = roi.transform.rotation;
334  output_mutex.lock();
335  aligned_objects->push_back(PoseEstimationMatch(
336  aligned_object, output_pose, object_roi_.dimensions, fitness));
337  output_mutex.unlock();
338  }

370 void PoseEstimator::NonMaxSuppression(
371  vector<PoseEstimationMatch>& aligned_objects,
372  vector<int>* deduped_indices) {
373  // TODO(jstn): remove nms_radius as a parameter. The radius should be
374  // dynamically sized based on the object dimensions. Instead of having just
375  // one radius, it should search for neighbors in an ellipsoid. Below, we are
376  // still using a single radius whose length is half of the longest dimension.
377  nms_radius_ =
378  std::max(std::max(object_dims_.x(), object_dims_.y()), object_dims_.z()) /
379  2.1;
380  ROS_INFO("Non-max suppression radius is %f", nms_radius_);
381 
382  deduped_indices->clear();
383  // Non-max supression of output objects
384  // Index centers into a tree
385  PointCloudP::Ptr output_centers(new PointCloudP);
386  PointCloudC::Ptr viz_deduped(new PointCloudC);
387  viz_deduped->header.frame_id = scene_->header.frame_id;
388  for (size_t i = 0; i < aligned_objects.size(); ++i) {
389  output_centers->push_back(aligned_objects[i].center());
390  }
391  PointPTree output_centers_tree;
392  output_centers_tree.setInputCloud(output_centers);
393 
394  // Suppress non max within a radius
395  for (int i = 0; i < static_cast<int>(aligned_objects.size()); ++i) {
396  PoseEstimationMatch& match = aligned_objects[i];
397  vector<int> neighbor_indices;
398  vector<float> neighbor_distances;
399  output_centers_tree.radiusSearch(match.center(), nms_radius_,
400  neighbor_indices, neighbor_distances);
401 
402  bool keep_match = true;
403  for (size_t k = 0; k < neighbor_indices.size(); ++k) {
404  int index = neighbor_indices[k];
405  if (index == i) {
406  continue;
407  }
408  PoseEstimationMatch& neighbor = aligned_objects[index];
409  if (neighbor.fitness() < match.fitness()) {
410  if (debug_) {
411  ROS_INFO("Suppressing match %d, %f >= neighbor %d, %f", i,
412  match.fitness(), index, neighbor.fitness());
413  }
414  keep_match = false;
415  break;
416  } else if (neighbor.fitness() == match.fitness()) {
417  // Keep this one, bump neighbor's fitness score up so that it won't come
418  // up again.
419  // neighbor.set_fitness(neighbor.fitness() + 0.0001);
420  keep_match = false;
421  }
422  }
423  if (keep_match) {
424  deduped_indices->push_back(i);
425  PointC max;
426  max.x = match.center().x;
427  max.y = match.center().y;
428  max.z = match.center().z;
429  viz_deduped->push_back(max);
430  }
431  }
432 
433  viz::PublishCloud(candidates_pub_, *viz_deduped);
434 }

436 void PoseEstimator::FilterMatches(
437  const vector<PoseEstimationMatch>& aligned_objects,
438  const vector<int>& deduped_indices, vector<int>* output_indices) {
439  double best_fitness = std::numeric_limits<double>::max();
440  int best_index = 0;
441  vector<double> scores(deduped_indices.size());
442  vector<pair<int, double> > index_scores(deduped_indices.size());
443  for (size_t di = 0; di < deduped_indices.size(); ++di) {
444  int index = deduped_indices[di];
445  const PoseEstimationMatch& match = aligned_objects[index];
446  double fitness = match.fitness();
447  scores[di] = fitness;
448  index_scores[di] = std::make_pair(index, fitness);
449  string threshold_marker = "";
450  if (fitness < fitness_threshold_) {
451  output_indices->push_back(index);
452  threshold_marker = " -- below threshold";
453  }
454  if (debug_) {
455  ROS_INFO("(%ld/%ld) ICP converged with score: %f%s", di + 1,
456  deduped_indices.size(), fitness, threshold_marker.c_str());
457  }
458 
459  if (fitness < best_fitness) {
460  best_fitness = fitness;
461  best_index = index;
462  }
463 
464  viz::PublishCloud(alignment_pub_, *(match.cloud()));
465  if (debug_ && match.fitness() < 0.01) {
466  std::cout << "Score: " << match.fitness() << ", press enter to continue ";
467  string user_input;
468  std::getline(std::cin, user_input);
469  }
470  }
471  ROS_INFO("Best score in candidates: %f", best_fitness);
472 
473  if (output_indices->size() == 0) {
474  // Some matches don't meet the threshold but are significantly better than
475  // the other matches. We look at the # of standard deviations from the mean
476  // to see if the best match should be output anyway.
477  double sum = std::accumulate(scores.begin(), scores.end(), 0.0);
478  double mean = sum / scores.size();
479  double sq_sum =
480  std::inner_product(scores.begin(), scores.end(), scores.begin(), 0.0);
481  double stdev = std::sqrt(sq_sum / scores.size() - mean * mean);
482 
483  double sigmas = (mean - best_fitness) / stdev;
484  if (sigmas >= sigma_threshold_) {
485  output_indices->push_back(best_index);
486  }
487  ROS_INFO("Fitness mean: %f stdev: %f, best is %f sigmas from mean", mean,
488  stdev, sigmas);
489  }
490 
491  if (static_cast<int>(output_indices->size()) < min_results_) {
492  int num_results =
493  std::min(min_results_, static_cast<int>(index_scores.size()));
494  std::partial_sort(index_scores.begin(), index_scores.begin() + num_results,
495  index_scores.end(), &CompareScores);
496  int start = static_cast<int>(output_indices->size());
497  for (int i = start; i < num_results; ++i) {
498  const pair<int, double>& index_score = index_scores[i];
499  output_indices->push_back(index_score.first);
500  }
501  }
502 }
