39  // Sample randomly
40  int num_samples = static_cast<int>(round(sample_ratio_ * scene_->size()));
41  num_samples = std::min(num_samples, max_samples_);
42  pcl::RandomSample<PointC> random;
43  random.setSeed(0);
44  random.setInputCloud(scene_);
45  random.setSample(num_samples);
46  random.filter(indices->indices);
47  ROS_INFO("Randomly sampled %ld points", indices->indices.size());
48 
49  PointCloudC::Ptr working_object(new PointCloudC);
50  PointCloudC::Ptr aligned_object(new PointCloudC);
51  importances->resize(indices->indices.size());
52  pcl::IterativeClosestPoint<PointC, PointC> icp;
53  icp.setMaximumIterations(0);
54  icp.setInputTarget(scene_);
55  for (size_t indices_i = 0; indices_i < indices->indices.size(); ++indices_i) {
56  int index = indices->indices[indices_i];
57  const PointC& center = scene_->points[index];
58  // Transform the object to be centered on the center point
59  // So far we just use a translation offset
60  // Also assume that the object and scene are in the same frame
61  Eigen::Vector3d translation;
62  translation.x() = center.x - object_center_.x();
63  translation.y() = center.y - object_center_.y();
64  translation.z() = center.z - object_center_.z();
65  pcl::transformPointCloud(*object_, *working_object, translation,
66  Eigen::Quaterniond::Identity());
67 
68  // Run ICP
69  icp.setInputSource(working_object);
70  icp.align(*aligned_object);
71 
72  rapid_msgs::Roi3D roi = object_roi_;
73  roi.transform.translation.x += translation.x();
74  roi.transform.translation.y += translation.y();
75  roi.transform.translation.z += translation.z();
76  double fitness = ComputeIcpFitness(scene_, aligned_object, roi);
77  (*importances)(indices_i) = fitness;
78 
79  // viz::PublishCloud(heatmap_pub_, *aligned_object);
80  // if (debug_) {
81  // std::cout << "Score: " << fitness;

87  double min = importances->minCoeff();
88  importances->array() -= min;

90  (*importances) /= max;

92 
93  // Color point cloud for visualization

96  for (size_t indices_i = 0; indices_i < indices->indices.size(); ++indices_i) {

98  int index = indices->indices[indices_i];

102  }
