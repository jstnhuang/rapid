116  Eigen::Matrix4f transform_mat = Eigen::Matrix4f::Zero();
117  transform_mat.topLeftCorner(3, 3) = rotation;
118  transform_mat(0, 3) = rt[9];
119  transform_mat(1, 3) = rt[10];

130  pose.orientation.z = quaternion.z();
131  pose.position.x = rt[9];
132  pose.position.y = rt[10];
133  pose.position.z = rt[11];
134  p_arr.poses.push_back(pose);
135  PoseEstimationMatch match(working_transformed, pose, object_roi_.dimensions,
136  it->match_confidence_);
