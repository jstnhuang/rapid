112 def MakeRelativePathsInFlagsAbsolute( flags, working_directory ):
113  if not working_directory:
114  return list( flags )
115  new_flags = []
116  make_next_absolute = False
117  path_flags = [ '-isystem', '-I', '-iquote', '--sysroot=' ]
118  for flag in flags:
119  new_flag = flag
120 
121  if make_next_absolute:
122  make_next_absolute = False
123  if not flag.startswith( '/' ):
124  new_flag = os.path.join( working_directory, flag )
125 
126  for path_flag in path_flags:
127  if flag == path_flag:
128  make_next_absolute = True
129  break
130 
131  if flag.startswith( path_flag ):
132  path = flag[ len( path_flag ): ]
133  new_flag = path_flag + os.path.join( working_directory, path )
134  break
135 
136  if new_flag:
137  new_flags.append( new_flag )
138  return new_flags

146 def GetCompilationInfoForFile( filename ):
147  # The compilation_commands.json file generated by CMake does not have entries
148  # for header files. So we do our best by asking the db for flags for a
149  # corresponding source file, if any. If one exists, the flags for that file
150  # should be good enough.
151  if IsHeaderFile( filename ):
152  basename = os.path.splitext( filename )[ 0 ]
153  for extension in SOURCE_EXTENSIONS:
154  replacement_file = basename + extension
155  if os.path.exists( replacement_file ):
156  compilation_info = database.GetCompilationInfoForFile(
157  replacement_file )
158  if compilation_info.compiler_flags_:
159  return compilation_info
160  return None
161  return database.GetCompilationInfoForFile( filename )
