45 bool Gripper::SetPosition(double position, double effort) {
46  if (position > Gripper::OPEN) {
47  ROS_WARN("Clamping gripper position %0.3f to max: %0.3f", position,
48  Gripper::OPEN);
49  position = Gripper::OPEN;
50  } else if (position < Gripper::CLOSED) {
51  ROS_WARN("Clamping gripper position %0.3f to min: %0.3f", position,
52  Gripper::CLOSED);
53  position = Gripper::CLOSED;
54  }
55  if (!gripper_client_->waitForServer(ros::Duration(5))) {
56  ROS_ERROR("The gripper action server was not available after 5 seconds!");
57  return false;
58  }
59 
60  double start_position = GetPosition();
61 
62  pr2_controllers_msgs::Pr2GripperCommandGoal goal;
63  goal.command.position = position;
64  goal.command.max_effort = effort;
65 
66  gripper_client_->sendGoal(goal);
67  if (!gripper_client_->waitForResult(ros::Duration(10))) {
68  ROS_ERROR(
69  "The gripper action server did not return a result after 10 seconds!");
70  return false;
71  }
72  SimpleClientGoalState state = gripper_client_->getState();
73  if (state == SimpleClientGoalState::SUCCEEDED ||
74  state == SimpleClientGoalState::ABORTED) {
75  if (position > start_position) {
76  // After the gripper opens, it's unlikely to hold an object.
77  is_holding_object_ = false;
78  }
79  return true;
80  } else {
81  ROS_ERROR("Gripper goal state: %s\n", state.toString().c_str());
82  return false;
83  }
84 }

86 double Gripper::GetPosition() const {
87  tf::StampedTransform transform;
88  std::string target_frame;
89  std::string source_frame;
90 
91  // set frames based on whether this is a right or left gripper
92  if (gripper_id_ == Gripper::LEFT_GRIPPER) {
93  source_frame = "l_gripper_l_finger_tip_link";
94  target_frame = "l_gripper_r_finger_tip_link";
95  } else if (gripper_id_ == Gripper::RIGHT_GRIPPER) {
96  source_frame = "r_gripper_l_finger_tip_link";
97  target_frame = "r_gripper_r_finger_tip_link";
98  }
99 
100  // get the transform between the fingertips
101  try {
102  transform_listener_->waitForTransform(target_frame, source_frame,
103  ros::Time(0), ros::Duration(10.0));
104  transform_listener_->lookupTransform(target_frame, source_frame,
105  ros::Time(0), transform);
106  } catch (tf::TransformException ex) {
107  ROS_ERROR("%s", ex.what());
108  return -1;
109  }
110 
111  // subtract small positive offset so that 0 means closed
112  double gripper_offset = transform.getOrigin().y();
113  ROS_INFO("gripper_offset = %f", gripper_offset);
114  return gripper_offset - 0.032;
115 }
