60  int* index, double* cosine_sim) {

62  double best_sim = 0; // Highest absolute value of cosine sim.

105 
106  // Compute midpoint, defined as the midpoint between the minimum and maximum
107  // points, in x, y, and z directions. The centroid is an average that depends
108  // on the density of points, which doesn't give you the geometric center of
109  // the point cloud.
110  PointXYZRGB eigen_center;
111  eigen_center.x = eigen_min.x + x_length / 2;
112  eigen_center.y = eigen_min.y + y_length / 2;
113  eigen_center.z = eigen_min.z + z_length / 2;
114  PointXYZRGB center;
115  pca.reconstruct(eigen_center, center);
116

118  double z_cosine_sim = 0;
119  int z_closest_index = 0;
120  ClosestAxis(eigenvectors, Eigen::Vector3d::UnitZ(), &z_closest_index,
121  &z_cosine_sim);
122  double x_cosine_sim = 0;
123  int x_closest_index = 0;
124  ClosestAxis(eigenvectors, Eigen::Vector3d::UnitX(), &x_closest_index,
125  &x_cosine_sim);
126

128  mygenvectors.col(0) = eigenvectors.col(x_closest_index);
129  if (x_cosine_sim < 0) {

131  }
132  mygenvectors.col(2) = eigenvectors.col(z_closest_index);
133  if (z_cosine_sim < 0) {

135  }

137

139  std::swap(x_length, z_length);
140  } else if (z_closest_index == 1) {
141  std::swap(y_length, z_length);
142  }
143 
144  if (x_closest_index == 1) {
145  std::swap(x_length, y_length);
146  } else if (x_closest_index == 2) {
147  std::swap(x_length, z_length);
148  }
149 
150  Eigen::Quaterniond q1(mygenvectors);
151 
152  // Output midpoint.
153  midpoint->header.frame_id = cloud->header.frame_id;
154  midpoint->pose.position.x = center.x;
155  midpoint->pose.position.y = center.y;
156  midpoint->pose.position.z = center.z;

267 double ComputeResolution(
268  const pcl::PointCloud<pcl::PointXYZRGB>::ConstPtr& cloud,
269  const pcl::search::Search<pcl::PointXYZRGB>& tree) {
270  double res = 0.0;
271  int n_points = 0;
272  int nres;
273  std::vector<int> indices(2);
274  std::vector<float> sqr_distances(2);
275 
276  for (size_t i = 0; i < cloud->size(); ++i) {
277  if (!pcl_isfinite((*cloud)[i].x)) {
278  continue;
279  }
280  // Considering the second neighbor since the first is the point itself.
281  nres = tree.nearestKSearch(i, 2, indices, sqr_distances);
282  if (nres == 2) {
283  res += sqrt(sqr_distances[1]);
284  ++n_points;
285  }
286  }
287  if (n_points != 0) {
288  res /= n_points;
289  }
290  return res;
291 }
