113  pose_.orientation.w = 1;
114 
115  // Compute center and size.
116  pcl::PointCloud<pcl::PointXYZRGB>::Ptr pcl_cloud(
117  new pcl::PointCloud<pcl::PointXYZRGB>);
118  pcl::fromROSMsg(cloud, *pcl_cloud);
119  Eigen::Vector4f min_pt;
120  Eigen::Vector4f max_pt;
121  pcl::getMinMax3D(*pcl_cloud, min_pt, max_pt);
122  geometry_msgs::Pose center;
123  center.orientation.w = 1;
124  center.position.x = (max_pt.x() + min_pt.x()) / 2;
125  center.position.y = (max_pt.y() + min_pt.y()) / 2;
126  center.position.z = (max_pt.z() + min_pt.z()) / 2;
127  double x_length = (max_pt.x() - min_pt.x());
128  double y_length = (max_pt.y() - min_pt.y());
129  double z_length = (max_pt.z() - min_pt.z());
130  max_dim_ = std::max(x_length, std::max(y_length, z_length));
131  graph_.Add("original", transform_graph::RefFrame(cloud.header.frame_id),
132  center);
133  pose_ = center;
134 
135  // Shift the cloud so that the center of its bounding box is at the origin.
136  Eigen::Vector4f centroid = (max_pt + min_pt) / 2;
137  pcl::demeanPointCloud(*pcl_cloud, centroid, *pcl_cloud);
138  pcl::toROSMsg(*pcl_cloud, cloud_);
