268 cv::Mat ReadMeanFile(const string& mean_file, const cv::Size& input_geometry,
269  int num_channels, string* error) {
270  caffe::BlobProto blob_proto;
271  caffe::ReadProtoFromBinaryFileOrDie(mean_file.c_str(), &blob_proto);
272 
273  /* Convert from BlobProto to Blob<float> */
274  caffe::Blob<float> mean_blob;
275  mean_blob.FromProto(blob_proto);
276  if (mean_blob.channels() != num_channels) {
277  if (error != NULL) {
278  *error = "Number of channels of mean file doesn't match input layer.";
279  }
280  }
281 
282  /* The format of the mean file is planar 32-bit float BGR or grayscale. */
283  vector<cv::Mat> channels;
284  float* data = mean_blob.mutable_cpu_data();
285  for (int i = 0; i < num_channels; ++i) {
286  /* Extract an individual channel. */
287  cv::Mat channel(mean_blob.height(), mean_blob.width(), CV_32FC1, data);
288  channels.push_back(channel);
289  data += mean_blob.height() * mean_blob.width();
290  }
291 
292  /* Merge the separate channels into a single image. */
293  cv::Mat mean;
294  cv::merge(channels, mean);
295 
296  /* Compute the global mean pixel value and create a mean image
297  * filled with this value. */
298  cv::Scalar channel_mean = cv::mean(mean);
299  mean = cv::Mat(input_geometry, mean.type(), channel_mean);
300  return mean;
301 }
